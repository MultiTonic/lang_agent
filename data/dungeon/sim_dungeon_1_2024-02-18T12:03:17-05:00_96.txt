<room_description> ::= "In this room, there is a(n) " <type_name> " named " <item_name> ", with " <properties> "."
                       {", and " <room_feature>}

<dungeon_feature> ::= (<monster> | <treasure> | <secret>)

<type_name> ::= "weapon" | "spell" | "artifact" | "magic item" | "treasure" | "monster" | "quest" | "challenge" | "theme" | "drama" | "music" | "art"

<item_name> ::= <alpha>+

<properties> ::= <property> {", " <property>}

<property> ::= <alpha>+ " of " <type_name>

<room_feature> ::= (<trap> | <puzzle> | <npc>)

<monster> ::= "There is a monster, the " <monster_name> ", in this room."

<treasure> ::= "There is a treasure chest containing " <treasure_contents> " in this room."

<secret> ::= "This room contains a secret that can be discovered by " <secret_condition> "."

<trap> ::= "There is a trap in this room, the " <trap_name> ", which can be disarmed by " <trap_disarm_condition> "."

<puzzle> ::= "There is a puzzle in this room, the " <puzzle_name> ", which can be solved by " <puzzle_solution_condition> "."

<npc> ::= "There is a non-player character, the " <npc_name> ", in this room. They will " <npc_action> " if you " <npc_trigger_condition> "."
