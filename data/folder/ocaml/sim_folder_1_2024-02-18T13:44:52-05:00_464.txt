type output = {

let merge_prompts (inputs : input list) (n : int) : output option =
  let rec aux acc inputs n =
    match inputs, n with
    | [], _ -> Some({result = acc; is_good = true})
    | _, 0 -> None
    | h::t, _ ->
      let new_acc = LanguageModel.generate_description ~grammar:h.grammar ~prev_results:h.prev_results in
      aux (new_acc :: acc) t (n - 1)
  in
  aux [] inputs n

let rec fold_left op acc = function
  | []   -> acc
  | h :: t -> fold_left op (op acc h) t

let rec play_game (inputs : input list) : unit =
