```ocaml
type input = Grammar of string | Previous of string | NewExample of string
type output = string

let rec merge_inputs acc = function
  | [] -> acc
  | Grammar a :: t -> merge_inputs (a :: acc) t
  | Previous d :: t -> merge_inputs (d :: acc) t
  | NewExample c :: t ->
      let merged_str = String.concat " " (List.rev acc) in
      let new_output = LanguageModel.generate ~grammar:merged_str ~previous:d ~new_example:c in
      if LanguageModel.test new_output then
        merge_inputs [new_output] t
      else merge_inputs (new_output :: acc) t

let rec generate_descriptions grammar previous_examples examples =
  let merged_inputs = List.map NewExample examples @ [Previous previous_examples; Grammar grammar] in
  merge_inputs [] merged_inputs
```

The `merge_inputs` function takes a list of strings representing different types of inputs (grammar, previous results, or new examples) and accumulates them into one string to generate the next output from the language model. If the generated output passes the test, it proceeds with the new output; otherwise, it continues to refine the current output up to 3 times.

The `generate_descriptions` function is a wrapper that initializes the input list with the given grammar, previous examples, and new examples, then starts merging them using the `merge_inputs` function.
